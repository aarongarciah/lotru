## Usage

```tsx
import { Field, FieldControl, FieldDescription, FieldLabel } from "@/components/ui/field";
```

```tsx
<Field>
  <FieldLabel>Name</FieldLabel>
  <FieldControl>
    <input type="text" />
  </FieldControl>
  <FieldDescription>We will use this name to personalize your experience.</FieldDescription>
</Field>
```
