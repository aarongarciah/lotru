## Usage

```tsx
import {
  Popover,
  PopoverArrow,
  PopoverContent,
  PopoverPositioner,
  PopoverTrigger,
} from "@/components/ui/popover";
```

```tsx
<Popover>
  <PopoverTrigger>Open popover</PopoverTrigger>
  <PopoverPositioner>
    <PopoverContent>
      <PopoverArrow />
      Place content for the popover here.
    </PopoverContent>
  </PopoverPositioner>
</Popover>
```
