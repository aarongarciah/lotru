## Installation

```bash
npx lotru@latest add ui/popover
```

## Usage

```tsx
import {
  Popover,
  PopoverArrow,
  PopoverContent,
  PopoverDescription,
  PopoverPositioner,
  PopoverTitle,
  PopoverTrigger,
} from "@/components/ui/popover";
```

```tsx
<Popover>
  <PopoverTrigger>Open popover</PopoverTrigger>
  <PopoverPositioner>
    <PopoverContent>
      <PopoverArrow />
      <PopoverTitle>Popover Title</PopoverTitle>
      <PopoverDescription>Popover Description</PopoverDescription>
    </PopoverContent>
  </PopoverPositioner>
</Popover>
```
